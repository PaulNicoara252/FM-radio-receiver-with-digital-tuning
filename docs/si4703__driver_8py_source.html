156 def process_rds(self):
157 """
158 Call this FREQUENTLY in your main loop (every 40ms or so).
159 It reads the registers and updates the internal text buffers.
160 """
161 self._read_registers()
162
163 # Check RDSR (RDS Ready) bit in STATUSRSSI (Bit 15)
164 if not (self.shadow_regs[self.STATUSRSSI] & 0x8000):
165 return # No new data
166
167 # Read blocks
168 block_b = self.shadow_regs[self.RDSB]
169 block_c = self.shadow_regs[self.RDSC]
170 block_d = self.shadow_regs[self.RDSD]
171
172 # Determine Group Type (Top 5 bits of Block B)
173 # 0x0A (00000) = Basic Tuning (Station Name)
174 # 0x2A (00100) = Radio Text
175 group_type = block_b >> 11
176
177 # --- Group 0A: Station Name (PS) ---
178 if group_type == 0:
179 # The index of the 2 characters is in the last 2 bits of Block B
180 idx = (block_b & 0x03) * 2
181
182 char1 = (block_d >> 8) & 0xFF
183 char2 = (block_d & 0xFF)
184
185 # Filter valid printable ASCII (32-126)
186 if 32 <= char1 <= 126: self.rds_station_name[idx] = chr(char1)
187 if 32 <= char2 <= 126: self.rds_station_name[idx+1] = chr(char2)
188
189 # Update String Buffer
190 self.rds_name_buffer = "".join(self.rds_station_name)
191
192 # --- Group 2A: Radio Text (RT) ---
193 elif group_type == 4: # 4 is 0b00100 (Group 2A)
194 # Index is bits 3:0 of Block B (0-15)
195 # Each index holds 4 chars (Total 64 chars)
196 idx = (block_b & 0x0F) * 4
197
198 chars = [
199 (block_c >> 8) & 0xFF,
200 (block_c & 0xFF),
201 (block_d >> 8) & 0xFF,
202 (block_d & 0xFF)
203 ]
204
205 for i in range(4):
206 if (idx + i) < 64:
207 if 32 <= chars[i] <= 126:
208 self.rds_radio_text[idx + i] = chr(chars[i])
209 elif chars[i] == 0x0D: # Carriage Return (End of text)
210 # Optional: Clean buffer after this point?
211 pass
212
213 self.rds_text_buffer = "".join(self.rds_radio_text).strip()
214
215 # IMPORTANT: We must wait 40ms before polling again to let the chip
216 # clear the buffer, or we will read the same block twice.
217 # However, in a main loop, the user delay usually handles this.
218
