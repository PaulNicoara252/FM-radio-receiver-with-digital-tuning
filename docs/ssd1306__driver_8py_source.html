43 # resolution and layout
44 SET_DISP_START_LINE | 0x00,
45 SET_SEG_REMAP | 0x01, # column addr 127 mapped to SEG0

63 SET_ENTIRE_ON, # output follows RAM contents
64 SET_NORM_INV, # not inverted

66 SET_CHARGE_PUMP,
